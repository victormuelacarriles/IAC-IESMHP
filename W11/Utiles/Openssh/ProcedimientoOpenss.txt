# Install the OpenSSH Client/server
Add-WindowsCapability -Online -Name OpenSSH.Client~~~~0.0.1.0
Add-WindowsCapability -Online -Name OpenSSH.Server~~~~0.0.1.0

# Start the sshd service
Set-Service -Name sshd -StartupType 'Automatic'
Start-Service sshd

# Confirm the Firewall rule is configured. It should be created automatically by setup. Run the following to verify
if (!(Get-NetFirewallRule -Name "OpenSSH-Server-In-TCP" -ErrorAction SilentlyContinue)) {
    Write-Output "Firewall Rule 'OpenSSH-Server-In-TCP' does not exist, creating it..."
    New-NetFirewallRule -Name 'OpenSSH-Server-In-TCP' -DisplayName 'OpenSSH Server (sshd)' -Enabled True -Direction Inbound -Protocol TCP -Action Allow -LocalPort 22
} else {
    Write-Output "Firewall rule 'OpenSSH-Server-In-TCP' has been created and exists."
}

#Ponemos por defecto powershell
New-ItemProperty -Path "HKLM:\SOFTWARE\OpenSSH" -Name DefaultShell -Value "C:\Windows\System32\WindowsPowerShell\v1.0\powershell.exe" -PropertyType String -Force



# 1. Definir la URL de donde se descargarán las claves
$url = "https://raw.githubusercontent.com/victormuelacarriles/IAC-IESMHP/refs/heads/main/Autorizados.txt"

# 2. Definir la ruta específica para administradores en Windows
$sshDir = "$env:ProgramData\ssh"
$authFile = "$sshDir\administrators_authorized_keys"

# 3. Asegurar que el directorio existe
if (-not (Test-Path $sshDir)) { 
    New-Item -ItemType Directory -Force -Path $sshDir | Out-Null
}

# 4. Descargar las claves desde GitHub y separar por saltos de línea
# Usamos un split por expresiones regulares para soportar tanto saltos de línea Linux (\n) como Windows (\r\n)
$nuevasClaves = (Invoke-RestMethod -Uri $url) -split '\r?\n' | Where-Object { $_.Trim() -ne "" }

# 5. Leer las claves que ya están configuradas en el equipo
$clavesExistentes = @()
if (Test-Path $authFile) {
    $clavesExistentes = Get-Content -Path $authFile | Where-Object { $_.Trim() -ne "" }
}

# 6. Insertar solo las claves que no estén ya en el archivo
$clavesAñadidas = 0
foreach ($clave in $nuevasClaves) {
    $claveLimpia = $clave.Trim()
    
    # Comprobamos si la clave exacta ya está en nuestro array de claves existentes
    if ($clavesExistentes -notcontains $claveLimpia) {
        Add-Content -Force -Path $authFile -Value $claveLimpia
        
        # Añadimos la clave a la lista en memoria por si el TXT de origen tuviera claves duplicadas
        $clavesExistentes += $claveLimpia 
        $clavesAñadidas++
    }
}

Write-Host "Proceso completado. Se han añadido $clavesAñadidas clave(s) nueva(s)." -ForegroundColor Green


#Permisos:
$authFile = "$env:ProgramData\ssh\administrators_authorized_keys"
icacls $authFile /inheritance:r
icacls $authFile /grant "*S-1-5-18:F"      # SYSTEM
icacls $authFile /grant "*S-1-5-32-544:F"  # Administradores (cualquier idioma)
